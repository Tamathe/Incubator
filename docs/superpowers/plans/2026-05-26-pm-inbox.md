# PM Inbox Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let team members log actions, blockers, decisions, and closure updates between Fridays via a public `/inbox` form, with a curator queue at `/admin`. GitHub Issues is the canonical store; the static site rebuilds on approval.

**Architecture:** A Cloudflare Worker mediates between the public form, GitHub Issues (the store), and a Vercel deploy hook. A build-time script pulls approved issues into `content/inbox.json`. A pure `mergeInbox()` function merges that with `content/site.ts` at build time. Direct PRs to `content/site.ts` continue to work.

**Tech Stack:** Next.js 15 (static export), TypeScript, Cloudflare Workers (Hono), GitHub Issues REST API, Cloudflare Turnstile, Vitest for `mergeInbox` tests.

      project?: string;
      question: string;

      targetId?: string;
      body: string;
    });

 * Field on an existing ActionItem | Blocker | Decision that an approved
 * update may rewrite. Curator picks this in /admin during approval.

  | "resolved"
  | "outcome"
  | "decidedAt"

  actions: ActionItem[];
  blockers: Blocker[];
  decisions: Decision[];
  updates: AppliedUpdate[];
}

```typescript
import type {
  AppliedUpdate,
  InboxData,
  UpdateField,
} from "@/lib/inbox-types";

/**
 * Merge curator-approved inbox items into a SiteContent. Returns a fresh
 * SiteContent — does not mutate the input. content/site.ts always wins
 * on id collisions; updates apply in chronological order of appliedAt.
 *
 * Unknown update targetIds are silently dropped (defensive — already
 * filtered upstream at build time).
 */
export function mergeInbox(content: SiteContent, inbox: InboxData): SiteContent {
  const existingActionIds = new Set(content.actions.map((a) => a.id));
  const existingBlockerIds = new Set(content.blockers.map((b) => b.id));
  const existingDecisionIds = new Set(content.decisions.map((d) => d.id));

  const mergedActions = [...content.actions];
  for (const a of inbox.actions) {
    if (existingActionIds.has(a.id)) {
      process.stderr.write(
        `[mergeInbox] duplicate action id "${a.id}" — content/site.ts wins, inbox dropped.\n`
      );
      continue;
    }
    mergedActions.push(a);
  }

  const mergedBlockers = [...content.blockers];
  for (const b of inbox.blockers) {
    if (existingBlockerIds.has(b.id)) {
      process.stderr.write(
        `[mergeInbox] duplicate blocker id "${b.id}" — content/site.ts wins, inbox dropped.\n`
      );
      continue;
    }
    mergedBlockers.push(b);
  }

  const mergedDecisions = [...content.decisions];
  for (const d of inbox.decisions) {
    if (existingDecisionIds.has(d.id)) {
      process.stderr.write(
        `[mergeInbox] duplicate decision id "${d.id}" — content/site.ts wins, inbox dropped.\n`
      );
      continue;
    }
    mergedDecisions.push(d);
  }

  const sortedUpdates = [...inbox.updates].sort((a, b) =>
    a.appliedAt < b.appliedAt ? -1 : a.appliedAt > b.appliedAt ? 1 : 0
  );

  for (const u of sortedUpdates) {
    if (!applyToCollection(mergedActions, u)) {
      if (!applyToCollection(mergedBlockers, u)) {
        applyToCollection(mergedDecisions, u);
      }
    }
  }

  return {
    ...content,
    actions: mergedActions,
    blockers: mergedBlockers,
    decisions: mergedDecisions,
  };
}

function applyToCollection<T extends { id: string }>(
  items: T[],
  update: AppliedUpdate
): boolean {
  const idx = items.findIndex((x) => x.id === update.targetId);
  if (idx === -1) return false;
  const next = { ...items[idx] } as Record<string, unknown>;
  next[update.field as UpdateField] = update.value;
  items[idx] = next as T;
  return true;
}
```

- [ ] **Step 1: Write the script**

Create `scripts/build-projects-snapshot.mjs`:

```javascript
#!/usr/bin/env node
/**
 * Build-time projects snapshot.
 *
 * Reads content/site.ts as text, extracts the project { id, name } pairs
 * via regex, and writes content/projects-snapshot.json which the
 * /inbox client form imports for the project dropdown.
 *
 * Regex parsing keeps this a zero-dependency .mjs script (matching the
 * existing scripts/build-changelog.mjs style). content/site.ts is
 * curator-edited and structurally stable enough that a regex over `id:`
 * and `name:` literals within the projects[] block is robust.
 */

import { readFileSync, writeFileSync } from "node:fs";
import { dirname, join } from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = join(__dirname, "..");
const SRC = join(ROOT, "content", "site.ts");
const OUT = join(ROOT, "content", "projects-snapshot.json");

function extractProjects(source) {
  const start = source.indexOf("projects: [");
  if (start === -1) return [];
  const end = source.indexOf("\n  ],", start);
  if (end === -1) return [];
  const block = source.slice(start, end);
  const projects = [];
  const objectRegex = /\{[^{}]*?id:\s*"([^"]+)"[^{}]*?name:\s*"([^"]+)"[^{}]*?\}/gs;
  let m;
  while ((m = objectRegex.exec(block)) !== null) {
    projects.push({ id: m[1], name: m[2] });
  }
  return projects;
}

function main() {

Create `scripts/build-inbox.mjs`:

```javascript
#!/usr/bin/env node
/**
 * Build-time inbox fetcher.
 *
 * Reads approved GitHub issues with labels `pm-inbox + approved`,
 * parses the fenced ```json block from each issue body, and writes
 * content/inbox.json — consumed at build time by lib/content.ts.
 *
 * Tolerates missing GH_TOKEN / GH_REPO env: writes an empty payload
 * and exits 0. `npm run build` never fails on inbox availability.
 */

import { writeFileSync } from "node:fs";
import { dirname, join } from "node:path";
import { fileURLToPath } from "node:url";

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = join(__dirname, "..");
const OUT = join(ROOT, "content", "inbox.json");

const TOKEN = process.env.GH_TOKEN;
const REPO = process.env.GH_REPO;
const MAX_ISSUES = 500;
const PER_PAGE = 100;

function writeEmpty(reason) {
  process.stderr.write(`[build-inbox] ${reason} — writing empty inbox.json\n`);
  writeFileSync(
    OUT,
    JSON.stringify(
      {
        generatedAt: new Date().toISOString(),
        actions: [],
        blockers: [],
        decisions: [],
        updates: [],
      },
      null,
      2
    )
  );
  process.exit(0);
}

async function fetchPage(page) {
  const url = `https://api.github.com/repos/${REPO}/issues?state=all&labels=pm-inbox,approved&per_page=${PER_PAGE}&page=${page}`;
  const res = await fetch(url, {
    headers: {
      Authorization: `Bearer ${TOKEN}`,
      Accept: "application/vnd.github+json",
      "X-GitHub-Api-Version": "2022-11-28",
      "User-Agent": "ai-incubator-build-inbox",
    },
  });
  if (!res.ok) {
    throw new Error(`GH API ${res.status}: ${await res.text()}`);
  }
  return res.json();
}

function extractJson(body) {
  if (!body) return null;
  const match = body.match(/```json\s*\n([\s\S]*?)\n```/);
  if (!match) return null;
  try {
    return JSON.parse(match[1]);
  } catch {
    return null;
  }
}

function typeLabelOf(labels) {
  for (const l of labels) {
    const name = typeof l === "string" ? l : l.name;
    if (name && name.startsWith("type:")) return name.slice(5);
  }
  return null;
}

async function main() {
  if (!TOKEN || !REPO) {
    return writeEmpty("GH_TOKEN/GH_REPO unset");
  }

  let issues = [];
  try {
    for (let page = 1; page <= Math.ceil(MAX_ISSUES / PER_PAGE); page++) {
      const batch = await fetchPage(page);
      if (!Array.isArray(batch) || batch.length === 0) break;
      issues = issues.concat(batch);
      if (batch.length < PER_PAGE) break;
    }
  } catch (err) {
    return writeEmpty(`fetch failed: ${err.message}`);
  }

  const actions = [];
  const blockers = [];
  const decisions = [];
  const updates = [];

  for (const issue of issues) {
    const data = extractJson(issue.body);
    if (!data) {
      process.stderr.write(
        `[build-inbox] issue #${issue.number}: no parseable JSON block, skipped.\n`
      );
      continue;
    }
    const type = typeLabelOf(issue.labels || []);
    if (type === "action") actions.push(data);
    else if (type === "blocker") blockers.push(data);
    else if (type === "decision") decisions.push(data);
    else if (type === "update") updates.push(data);
    else {
      process.stderr.write(
        `[build-inbox] issue #${issue.number}: unknown type label, skipped.\n`
      );
    }
  }

  writeFileSync(
    OUT,
    JSON.stringify(
      {
        generatedAt: new Date().toISOString(),
        actions,
        blockers,
        decisions,
        updates,
      },
      null,
      2
    )
  );
  process.stdout.write(
    `[build-inbox] wrote ${actions.length} actions, ${blockers.length} blockers, ${decisions.length} decisions, ${updates.length} updates to ${OUT}\n`
  );
}

main();
```

    }
  }

  if (authError) {
    return (
      <div className="card" style={{ padding: 32, maxWidth: 540 }}>
        <h2 className="h2">Admin queue</h2>
        <p className="body" style={{ marginTop: 12 }}>{authError}</p>
      </div>
    );
  }
  if (items === null) {
    return <div className="body">Loading…</div>;
  }
  if (items.length === 0) {
    return (
      <div className="card" style={{ padding: 32 }}>
        <h2 className="h2">Queue is empty</h2>
        <p className="body" style={{ marginTop: 12 }}>
          Nothing pending. New submissions will appear here.
        </p>
      </div>
    );
  }

  return (
    <div style={{ display: "grid", gap: 18 }}>
      {actionMessage && (
        <div className="card" style={{ padding: 14, fontSize: 13 }}>
          {actionMessage}
        </div>
      )}
      {items.map((item) => (
        <PendingRow
          key={item.issueNumber}
          item={item}
          onSend={sendApprove}
        />
      ))}
    </div>
  );
}

function PendingRow({
  item,
  onSend,
}: {
  item: PendingIssue;
  onSend: (
    issueNumber: number,
    action: "approve" | "reject" | "skip",
    payload: Record<string, unknown> | null
  ) => Promise<void>;
}) {
  const [draft, setDraft] = useState<Record<string, unknown>>(() => {
    if (item.type === "update") {
      return { targetId: "", field: "status", value: "", ...item.raw };
    }
    return { ...item.raw };
  });

  function set<K extends string>(key: K, value: unknown) {
    setDraft((d) => ({ ...d, [key]: value }));
  }

  function buildApprovedPayload(): Record<string, unknown> {
    const now = new Date().toISOString();
    const submittedAt = String(draft.submittedAt ?? now);
    const createdDate = submittedAt.slice(0, 10);

    if (item.type === "action") {
      return {
        id: String(draft.id ?? ""),
        project: String(draft.project ?? ""),
        owner: String(draft.owner ?? ""),
        body: String(draft.body ?? ""),
        created: createdDate,
        due: draft.due ? String(draft.due) : undefined,
        status: String(draft.status ?? "open"),
        closedAt: draft.closedAt ? String(draft.closedAt) : undefined,
      };
    }
    if (item.type === "blocker") {
      return {
        id: String(draft.id ?? ""),
        project: String(draft.project ?? ""),
        body: String(draft.body ?? ""),
        blockedBy: draft.blockedBy ? String(draft.blockedBy) : undefined,
        created: createdDate,
        resolved: draft.resolved ? String(draft.resolved) : undefined,
      };
    }
    if (item.type === "decision") {
      return {
        id: String(draft.id ?? ""),
        project: draft.project ? String(draft.project) : undefined,
        question: String(draft.question ?? ""),
        created: createdDate,
        forSession: String(draft.forSession ?? ""),
        status: String(draft.status ?? "queued"),
        outcome: draft.outcome ? String(draft.outcome) : undefined,
        decidedAt: draft.decidedAt ? String(draft.decidedAt) : undefined,
      };
    }
    // update
    return {
      targetId: String(draft.targetId ?? ""),
      field: String(draft.field ?? "status"),
      value: String(draft.value ?? ""),
      appliedAt: now,
    };
  }

  return (

      <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 8 }}>
        <span className="mono" style={{ fontSize: 12, color: "var(--ink-3)" }}>

        <ActionFields draft={draft} set={set} />

        <BlockerFields draft={draft} set={set} />

        <DecisionFields draft={draft} set={set} />
      )}
      {item.type === "update" && (
        <UpdateFields draft={draft} set={set} />
      )}

        <button
          className="btn primary"
          onClick={() => onSend(item.issueNumber, "approve", buildApprovedPayload())}
        >
          Approve
        </button>
        <button
          className="btn"
          onClick={() => onSend(item.issueNumber, "reject", null)}
        >
          Reject
        </button>
        <button
          className="btn ghost"

const recentRequests = new Map<string, number[]>();
function rateLimit(ip: string): boolean {
  const now = Date.now();
  const recent = (recentRequests.get(ip) ?? []).filter((t) => now - t < 3_600_000);
  if (recent.length >= 10) return false;
  recent.push(now);
  recentRequests.set(ip, recent);
  return true;
}

async function verifyTurnstile(secret: string, token: string, ip: string): Promise<boolean> {
  const res = await fetch("https://challenges.cloudflare.com/turnstile/v0/siteverify", {
    method: "POST",
    headers: { "Content-Type": "application/x-www-form-urlencoded" },
    body: new URLSearchParams({ secret, response: token, remoteip: ip }),
  });
  if (!res.ok) return false;
  const json = (await res.json()) as { success?: boolean };
  return Boolean(json.success);
}

function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, "-")
    .replace(/^-+|-+$/g, "")
    .slice(0, 40);
}

function randomSuffix(): string {
  return Math.random().toString(36).slice(2, 6);
}

function generateId(s: Submission): string {
  let seed = "";
  if (s.kind === "action") seed = `${s.project}-${s.body}`;
  else if (s.kind === "blocker") seed = `${s.project}-${s.body}`;
  else if (s.kind === "decision") seed = `${s.project ?? "x"}-${s.question}`;
  else seed = `update-${s.targetId ?? "x"}-${s.body}`;
  return `${slugify(seed) || "item"}-${randomSuffix()}`;
}

function titleFor(s: Submission, id: string): string {
  const projectPart =
    "project" in s && s.project ? s.project : "—";
  const previewBody =
    "body" in s ? s.body : "question" in s ? s.question : id;
  const truncated = previewBody.slice(0, 60);
  return `[pending] [${s.kind}] [${projectPart}] ${truncated}`;
}

function humanSummaryFor(s: Submission, id: string, submittedAt: string): string {
  const submitter = s.submitter?.name ? ` — submitted by ${s.submitter.name}` : "";
  if (s.kind === "action") {
    const due = s.due ? ` · due **${s.due}**` : "";
    return `**Action** for **${s.project}** · owner **${s.owner}**${due}\n\n${s.body}${submitter}`;
  }
  if (s.kind === "blocker") {
    const blockedBy = s.blockedBy ? ` · blocked by **${s.blockedBy}**` : "";
    return `**Blocker** for **${s.project}**${blockedBy}\n\n${s.body}${submitter}`;
  }
  if (s.kind === "decision") {
    const proj = s.project ? `for **${s.project}**` : "_cross-cutting_";
    return `**Decision** ${proj} · for session **${s.forSession}**\n\n${s.question}${submitter}`;
  }
  const target = s.targetId ? `target **${s.targetId}**` : "_no target id given_";
  return `**Update** · ${target}\n\n${s.body}${submitter}\n\n_(id: ${id}, submitted ${submittedAt})_`;
}

function bodyFor(s: Submission, id: string, submittedAt: string): string {
  const payload = { ...s, id, submittedAt };
  return [
    "```json",
    JSON.stringify(payload, null, 2),
    "```",
    "",
    humanSummaryFor(s, id, submittedAt),
  ].join("\n");
}

async function createIssue(
  env: Bindings,
  s: Submission,
  id: string,
  submittedAt: string

  const body = bodyFor(s, id, submittedAt);
  const labels = ["pm-inbox", "pending", `type:${s.kind}`];
  const res = await fetch(`https://api.github.com/repos/${env.GH_REPO}/issues`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${env.GH_TOKEN}`,
      Accept: "application/vnd.github+json",
      "X-GitHub-Api-Version": "2022-11-28",
      "User-Agent": "ai-incubator-inbox-worker",
      "Content-Type": "application/json",

    body: JSON.stringify({ title, body, labels }),
  });
  if (!res.ok) {
    throw new Error(`GH issue create ${res.status}: ${await res.text()}`);
  }
  const json = (await res.json()) as { number: number };
  return json.number;
}

  const verified = await verifyTurnstile(
    c.env.TURNSTILE_SECRET,
    submission.turnstileToken,
    ip
  );
  if (!verified) {
    return c.json({ error: "verification failed" }, 400);
  }

  const id = generateId(submission);
  const submittedAt = new Date().toISOString();

  try {
    const issueNumber = await createIssue(c.env, submission, id, submittedAt);
    return c.json({ ok: true, issue: issueNumber });
  } catch (err) {
    return c.json(
      { error: err instanceof Error ? err.message : "Inbox unreachable" },
      502
    );
  }

interface PendingIssue {

function requireAdmin(c: { req: { query: (k: string) => string | undefined }; env: Bindings }): boolean {
  const token = c.req.query("token");
  return Boolean(token) && token === c.env.ADMIN_TOKEN;
}

function extractJsonBlock(body: string | null | undefined): Record<string, unknown> | null {
  if (!body) return null;
  const match = body.match(/```json\s*\n([\s\S]*?)\n```/);
  if (!match) return null;
  try {
    return JSON.parse(match[1]) as Record<string, unknown>;
  } catch {
    return null;
  }
}

function summaryFromBody(body: string | null | undefined): string {
  if (!body) return "";
  const withoutJson = body.replace(/```json\s*\n[\s\S]*?\n```\s*/m, "").trim();
  return withoutJson.slice(0, 400);
}

function typeFromLabels(labels: Array<string | { name?: string }>): SubmissionKind | null {
  for (const l of labels) {
    const name = typeof l === "string" ? l : l.name ?? "";
    if (name.startsWith("type:")) {
      const t = name.slice(5);
      if (t === "action" || t === "blocker" || t === "decision" || t === "update") {
        return t;
      }
    }
  }
  return null;
}

async function fetchIssues(env: Bindings, labels: string): Promise<Array<{
  number: number;
  body: string | null;
  labels: Array<string | { name?: string }>;
  created_at: string;
}>> {
  const url = `https://api.github.com/repos/${env.GH_REPO}/issues?state=open&labels=${encodeURIComponent(labels)}&per_page=100`;
  const res = await fetch(url, {
    headers: {
      Authorization: `Bearer ${env.GH_TOKEN}`,
      Accept: "application/vnd.github+json",
      "X-GitHub-Api-Version": "2022-11-28",
      "User-Agent": "ai-incubator-inbox-worker",
    },
  });
  if (!res.ok) {
    throw new Error(`GH issues list ${res.status}: ${await res.text()}`);
  }
  return res.json();
}

app.get("/queue", async (c) => {
  if (!requireAdmin(c)) return c.json({ error: "unauthorized" }, 401);
  try {
    const issues = await fetchIssues(c.env, "pm-inbox,pending");
    const pending: PendingIssue[] = [];
    for (const issue of issues) {
      const raw = extractJsonBlock(issue.body);
      const type = typeFromLabels(issue.labels);
      if (!raw || !type) continue;
      pending.push({
        issueNumber: issue.number,
        type,
        createdAt: issue.created_at,
        raw,
        summary: summaryFromBody(issue.body),
      });
    }
    return c.json({ ok: true, pending });
  } catch (err) {
    return c.json(
      { ok: false, error: err instanceof Error ? err.message : "Queue unavailable" },
      502
    );
  }
});

```typescript
type ApproveAction = "approve" | "reject" | "skip";

interface ApproveBody {
  issueNumber: number;
  action: ApproveAction;
  payload?: Record<string, unknown> | null;
}

async function patchIssue(
  env: Bindings,
  issueNumber: number,
  patch: { body?: string; labels?: string[]; state?: "open" | "closed" }
): Promise<void> {
  const res = await fetch(
    `https://api.github.com/repos/${env.GH_REPO}/issues/${issueNumber}`,
    {
      method: "PATCH",
      headers: {
        Authorization: `Bearer ${env.GH_TOKEN}`,
        Accept: "application/vnd.github+json",
        "X-GitHub-Api-Version": "2022-11-28",
        "User-Agent": "ai-incubator-inbox-worker",
        "Content-Type": "application/json",
      },
      body: JSON.stringify(patch),
    }
  );
  if (res.status === 404) {
    throw Object.assign(new Error("Unknown issue"), { httpStatus: 404 });
  }
  if (!res.ok) {
    throw new Error(`GH issue patch ${res.status}: ${await res.text()}`);
  }
}

async function getIssue(
  env: Bindings,
  issueNumber: number
): Promise<{
  body: string | null;
  labels: Array<string | { name?: string }>;
}> {
  const res = await fetch(
    `https://api.github.com/repos/${env.GH_REPO}/issues/${issueNumber}`,
    {
      headers: {
        Authorization: `Bearer ${env.GH_TOKEN}`,
        Accept: "application/vnd.github+json",
        "X-GitHub-Api-Version": "2022-11-28",
        "User-Agent": "ai-incubator-inbox-worker",
      },
    }
  );
  if (res.status === 404) {
    throw Object.assign(new Error("Unknown issue"), { httpStatus: 404 });
  }
  if (!res.ok) {
    throw new Error(`GH issue get ${res.status}: ${await res.text()}`);
  }
  return res.json();
}

function rewriteLabels(
  existing: Array<string | { name?: string }>,
  next: "approved" | "rejected"
): string[] {
  const out: string[] = [];
  for (const l of existing) {
    const name = typeof l === "string" ? l : l.name ?? "";
    if (name === "pending" || name === "approved" || name === "rejected") continue;
    if (name) out.push(name);
  }
  out.push(next);
  return out;
}

function approvedBody(
  type: SubmissionKind,
  payload: Record<string, unknown>,
  originalSummary: string
): string {
  const json = JSON.stringify(payload, null, 2);
  return [
    "```json",
    json,
    "```",
    "",
    `**Approved ${type}** at ${new Date().toISOString()}`,
    "",
    "_Original submission summary preserved below:_",
    "",
    originalSummary,
  ].join("\n");
}

async function triggerDeploy(url: string): Promise<boolean> {
  try {
    const res = await fetch(url, { method: "POST" });
    return res.ok;
  } catch {
    return false;
  }
}

app.post("/approve", async (c) => {
  if (!requireAdmin(c)) return c.json({ error: "unauthorized" }, 401);

  let body: ApproveBody;
  try {
    body = await c.req.json<ApproveBody>();
  } catch {
    return c.json({ error: "Invalid JSON body" }, 400);
  }

  const { issueNumber, action } = body;
  if (!issueNumber || !action) {
    return c.json({ error: "issueNumber and action required" }, 400);
  }

  try {
    if (action === "skip") {
      return c.json({ ok: true });
    }

    if (action === "reject") {
      const issue = await getIssue(c.env, issueNumber);
      await patchIssue(c.env, issueNumber, {
        labels: rewriteLabels(issue.labels, "rejected"),
        state: "closed",
      });
      return c.json({ ok: true });
    }

    // approve
    if (!body.payload) {
      return c.json({ error: "payload required for approve" }, 400);
    }
    const issue = await getIssue(c.env, issueNumber);
    const type = typeFromLabels(issue.labels);
    if (!type) {
      return c.json({ error: "Issue missing type label" }, 400);
    }
    const summary = summaryFromBody(issue.body);
    await patchIssue(c.env, issueNumber, {
      body: approvedBody(type, body.payload, summary),
      labels: rewriteLabels(issue.labels, "approved"),
    });
    const deployOk = await triggerDeploy(c.env.DEPLOY_HOOK_URL);
    return c.json({ ok: true, deployHookFailed: !deployOk });
  } catch (err) {
    const status =
      err && typeof err === "object" && "httpStatus" in err
        ? (err as { httpStatus: number }).httpStatus
        : 502;
    return c.json(
      { ok: false, error: err instanceof Error ? err.message : "Approve failed" },
      status as 400 | 401 | 404 | 502
    );
  }
});
```
